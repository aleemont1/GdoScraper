 142    def fetch_promotions(self, store_id: str) -> Any:
 143        """
 144        Locates target PDF catalog files, either dynamically downloading them via REST 
 145        or scanning the local filesystem.
 146        """
 147        is_pdf = store_id.lower().endswith(".pdf")
 148        is_special = store_id.lower() in ("all", "downloads")
 149
 150        if not is_pdf and not is_special:
 151            logger.info(f"Checking for dynamic flyer downloads for store reference: '{store_id}'...")
 152            downloaded_paths = self.download_flyers(store_id)
 153            if downloaded_paths:
 154                logger.info(f"REST Downloader retrieved {len(downloaded_paths)} flyers.")
 155                return downloaded_paths
 156            else:
 157                logger.warning("No flyers retrieved via REST. Trying filesystem scan as fallback...")
 158
 159        pdf_paths: List[str] = []
 160        downloads_dir = self._download_subdir
 161        
 162        if store_id.lower() in ("all", "downloads"):
 163            search_pattern = os.path.join(downloads_dir, "*.pdf")
 164            pdf_paths = glob.glob(search_pattern)
 165            logger.info(f"Scanning downloads folder '{downloads_dir}'. Found {len(pdf_paths)} PDF flyers to scrape.")
 166            
 167        elif store_id.endswith(".pdf"):
 168            if os.path.exists(store_id):
 169                pdf_paths = [store_id]
 170            else:
 171                fallback_path = os.path.join(downloads_dir, os.path.basename(store_id))
 172                if os.path.exists(fallback_path):
 173                    pdf_paths = [fallback_path]
 174                else:
 175                    logger.error(f"Target PDF file not found at: {store_id}")
 176        else:
 177            search_pattern = os.path.join(downloads_dir, f"*{store_id}*.pdf")
 178            pdf_paths = glob.glob(search_pattern)
 179            if not pdf_paths:
 180                logger.error(f"No PDF flyer matching store ID: '{store_id}' found in '{downloads_dir}'.")
 181                
 182        return pdf_paths

 184    def parse_promotions(self, raw_data: Any, store_id: str) -> List[ProductOffer]:
 185        """
 186        Ingests the target PDF files and executes layout segmentation page-by-page,
 187        normalizing extracted text blocks into product offers with visual previews.
 188        Supports sequential and parallel multiprocessing modes.
 189        """
 190        if not isinstance(raw_data, list):
 191            logger.error("Invalid raw data structure provided. Expected a list of file paths.")
 192            return []
 193            
 194        # Determine store ID, filtering out any raw file path strings
 195        active_store_id = self._resolved_store_id
 196        if not active_store_id or active_store_id.endswith(".pdf") or "/" in active_store_id or "\\" in active_store_id:
 197            active_store_id = store_id
 198            
 199        if not active_store_id or active_store_id.endswith(".pdf") or "/" in active_store_id or "\\" in active_store_id:
 200            active_store_id = "MANUAL_STORE"
 201            
 202        logger.info(f"Using store ID: '{active_store_id}' for database and image tagging.")
 203
 204        all_parsed_offers: List[ProductOffer] = []
 205        
 206        if self.parallel and len(raw_data) > 1:
 207            import multiprocessing
 208            from concurrent.futures import ProcessPoolExecutor
 209            
 210            max_workers = min(len(raw_data), multiprocessing.cpu_count())
 211            logger.info(f"Initiating MULTIPROCESS PARALLEL parsing on {max_workers} processes (one process per flyer)...")
 212            
 213            driver_class = self.__class__
 214            with ProcessPoolExecutor(max_workers=max_workers) as executor:
 215                futures = [
 216                    executor.submit(_parse_single_flyer_worker, driver_class, file_path, active_store_id)
 217                    for file_path in raw_data
 218                ]
 219                
 220                for idx, future in enumerate(futures):
 221                    try:
 222                        offers = future.result()
 223                        all_parsed_offers.extend(offers)
 224                        logger.info(f"Parallel Worker #{idx+1} finished. Extracted {len(offers)} offers.")
 225                    except Exception as err:
 226                        logger.error(f"Parallel worker #{idx+1} failed with exception: {err}", exc_info=True)
 227        else:
 228            logger.info("Initiating SEQUENTIAL flyer parsing...")
 229            for file_path in raw_data:
 230                offers = self._parse_single_flyer_file(file_path, active_store_id)
 231                all_parsed_offers.extend(offers)
 232                
 233        logger.info(f"ETL Scrape completed. Extracted a total of {len(all_parsed_offers)} {self._supermarket_name} products.")
 234        return all_parsed_offers

 236    def _parse_single_flyer_file(self, file_path: str, store_id: str) -> List[ProductOffer]:
 237        """
 238        Core dynamic spatial ETL logic for parsing a single flyer PDF.
 239        Automatically detects vector vs scanned layout structure and routes accordingly.
 240        """
 241        if not os.path.exists(file_path):
 242            return []
 243            
 244        logger.info(f"Beginning spatial ETL pipeline on flyer: {os.path.basename(file_path)}")
 245        parsed_offers: List[ProductOffer] = []
 246        total_pages = 0
 247        has_vector_text = False
 248        
 249        try:
 250            with pdfplumber.open(file_path) as pdf:
 251                total_pages = len(pdf.pages)
 252                logger.info(f"Flyer has {total_pages} pages.")
 253                
 254                # Dynamic Vector vs Scanned Layout Detection
 255                if self._current_is_vector is not None:
 256                    has_vector_text = self._current_is_vector
 257                elif self.engine != "AUTO":
 258                    has_vector_text = False
 259                else:
 260                    has_vector_text = any(len(p.extract_words()) > 0 for p in pdf.pages)
 261                
 262                # --- STRATEGY A: Vector-Based PDF Parsing ---
 263                if has_vector_text:
 264                    logger.info("Vector-based PDF circular detected. Engaging high-accuracy text grid segmenter...")
 265                    
 266                    # 1. Extract Validity String from Page 1
 267                    validity_string = None
 268                    if total_pages > 0:
 269                        first_page = pdf.pages[0]
 270                        first_page_words = first_page.extract_words()
 271                        if first_page_words:
 272                            sorted_words = sorted(first_page_words, key=lambda w: (round(w["top"]/5)*5, w["x0"]))
 273                            first_page_text = " ".join([w["text"] for w in sorted_words])
 274                            validity_string = self._parser.parse_flyer_validity(first_page_text)
 275                            if validity_string:
 276                                logger.info(f"Flyer validity successfully resolved: '{validity_string}'")
 277                                
 278                    # 2. Segment and parse page cells
 279                    flyer_offers_count = 0
 280                    for page_idx in range(total_pages):
 281                        page = pdf.pages[page_idx]
 282                        
 283                        cells = self._segmenter.segment_page(page)
 284                        if not cells:
 285                            continue
 286                            
 287                        rendered_page_img = None
 288                        page_offers = []
 289                        
 290                        for cell in cells:
 291                            try:
 292                                offer = self._parser.parse_cell(cell["text"], store_id, validity_string)
 293                                if offer:
 294                                    # Instantiate card rendering on-demand
 295                                    if rendered_page_img is None:
 296                                        try:
 297                                            rendered_page_img = page.to_image(resolution=120).original
 298                                        except Exception as render_err:
 299                                            logger.debug(f"Failed to render page image: {render_err}")
 300                                            rendered_page_img = False
 301                                            
 302                                    if rendered_page_img and rendered_page_img is not False:
 303                                        offer.image_url = self._crop_and_save_card_image_from_cached(
 304                                            rendered_page_img, 
 305                                            cell["bbox"], 
 306                                            page, 
 307                                            store_id, 
 308                                            offer.offer_id,
 309                                            col_idx=cell.get("col_idx"),
 310                                            col_count=cell.get("col_count"),
 311                                            product_name=offer.name
 312                                        )
 313                                        
 314                                    offer.supermarket = self._supermarket_name
 315                                    page_offers.append(offer)
 316                            except ValueError as parse_err:
 317                                logger.debug(f"Cell parsing ValueError: {parse_err}")
 318                                cell_text = cell["text"]
 319                                if any(kw in cell_text.lower() for kw in ["€", "pezzo", "pezzi", "anziché"]):
 320                                    self._log_missed_product(
 321                                        file_path=file_path,
 322                                        page_idx=page_idx,
 323                                        reason=str(parse_err),
 324                                        text=cell_text
 325                                    )
 326                            except Exception as parse_err:
 327                                logger.debug(f"Cell parsing unexpected exception: {parse_err}")
 328                                
 329                        if self.use_claude and os.environ.get("ANTHROPIC_API_KEY"):
 330                            if rendered_page_img is None or rendered_page_img is False:
 331                                try:
 332                                    rendered_page_img = page.to_image(resolution=120).original
 333                                except Exception as render_err:
 334                                    logger.error(f"Failed to render page image for Claude visual audit: {render_err}")
 335                                    rendered_page_img = False
 336                                    
 337                            if rendered_page_img and rendered_page_img is not False:
 338                                page_offers = self._audit_and_self_heal_page_via_claude(
 339                                    rendered_page_img,
 340                                    store_id,
 341                                    page_idx,
 342                                    page_offers,
 343                                    validity_string
 344                                )
 345                                
 346                        for offer in page_offers:
 347                            parsed_offers.append(offer)
 348                            flyer_offers_count += 1
 349                            
 350                    logger.info(f"Finished parsing vector flyer {os.path.basename(file_path)}. Extracted {flyer_offers_count} products.")
 351                
 352                # --- STRATEGY B: Scanned PDF / Image-Only Brochure Parsing ---
 353                else:
 354                    logger.info("Scanned/Flat-image brochure detected. Engaging OCR visual extraction fallback...")
 355                    
 356                    # Unified system/visual prompts in English
 357                    prompt = (
 358                        "Role: You are an expert visual data and OCR extractor for Italian GDO (Grande Distribuzione Organizzata) promotional flyers.\n"
 359                        "Task: Carefully analyze the image of this flyer page and accurately extract all commercial offers.\n\n"
 360                        "Steps for each offer (step-by-step reasoning):\n"
 361                        "1. Visually locate a specific product promotional card/section.\n"
 362                        "2. Read the price (e.g., '1,49' -> convert to float 1.49) and any discount (e.g., '-30%'). If the price has euros and cents separated or minimized, combine them correctly.\n"
 363                        "3. Identify the descriptive text of the product in Italian, isolating the name (name), the brand (brand, e.g., 'Mulino Bianco') and the format/weight (weight_or_volume, e.g., '400 g', '1,5 L', 'pack of 4 pieces').\n"

 366                        "Validation Rules:\n"
 367                        "- Product Name (name): Write it in clean Title Case in Italian (e.g., 'Biscotti frollini integrali'). Do not include weights or brands here.\n"
 368                        "- Brand (brand): Extract only the actual brand (if specified). If absent, leave null.\n"

 371                    )
 372                    
 373                    # Engine B.1: Gemini Multimodal Visual OCR
 374                    if self.use_gemini:
 375                        model_name = os.environ.get("GEMINI_MODEL_NAME", "gemini-2.5-pro")
 376                        logger.info(f"Using Gemini {model_name} visual parsing API...")

 389

 393                            
 394                            try:
 395                                pil_img = page.to_image(resolution=120).original
 396                            except Exception as render_err:
 397                                logger.error(f"Failed to render page {page_idx + 1}: {render_err}")
 398                                continue
 399                                
 400                            try:
 401                                response = None
 402                                max_retries = 3
 403                                retry_backoff = 3
 404                                for attempt in range(max_retries):
 405                                    try:
 406                                        response = client.models.generate_content(
 407                                            model=model_name,
 408                                            contents=[pil_img, prompt],
 409                                            config=types.GenerateContentConfig(

 412                                                temperature=0.1

 431                                    name = o.get("name")

 438                                    orig_price = o.get("original_price")

 440                                    bbox = o.get("bbox")
 441                                    category = o.get("category")
 442

 629    def _crop_and_save_card_image_from_cached(

 634        store_id: str, 
 635        offer_id: str,
 636        col_idx: Optional[int] = None,

 643        """

 661

 663            return db_url
 664            
 665        try:
 666            img_w, img_h = pil_img.size
 667            page_w = float(page.width)
 668            page_h = float(page.height)
 669            
 670            scale_x = img_w / page_w
 671            scale_y = img_h / page_h
 672

 675                grid_x0 = bbox[0]
 676                grid_x1 = bbox[2]

 680                grid_x1 = (col_idx + 1) * col_w
 681            else:
 682                grid_x0 = 0
 683                grid_x1 = page_w

 712                ix1 = min(page_w, best_img["x1"] + pad)

 716                x1_px = int(ix1 * scale_x)
 717                top_px = int(iy0 * scale_y)
 718                bottom_px = int(iy1 * scale_y)

 721                    max(0, x0_px),
 722                    max(0, top_px),
 723                    min(img_w, x1_px),
 724                    min(img_h, bottom_px)
 725                )

 728                    x0 = bbox[0]
 729                    x1 = bbox[2]
 730                elif col_idx is not None and col_count is not None and col_count > 0:
 731                    col_w = page_w / col_count
 732                    x0 = col_idx * col_w
 733                    x1 = (col_idx + 1) * col_w
 734                else:

 739                w = x1 - x0

 741                    x0 = max(0.0, bbox[0] - 130)
 742                    x1 = min(page_w, bbox[2] + 15)

 754                    max(0, top_px - padding_top),
 755                    min(img_w, x1_px),
 756                    min(img_h, bottom_px + padding_bottom)
 757                )
