443        """Helper method to extract the first value of a property from the list by its code."""

453    def _parse_single_product(self, item: Dict[str, Any], store_id: str) -> Optional[ProductOffer]:
454        """Converts a single Dpiù product JSON element into a standard validated ProductOffer model."""
455        description = item.get("description", "") or ""
456        properties = item.get("properties", [])
457        
458        title = self._parse_property(properties, "TITLE") or ""
459        brand = self._parse_property(properties, "BRANDS") or title
460        
461        clean_desc = " ".join(description.split())
462        
463        if not clean_desc:
464            logger.debug("Skipping product missing description.")
465            return None
466            
467        ean_code = self._parse_property(properties, "EAN")
468        if ean_code:
469            ean_code = str(ean_code).strip()
470            
471        weight_str = None
472        weight_match = self._weight_regex.search(clean_desc)
473        if weight_match:
474            weight_str = weight_match.group(1).strip()
475        else:
476            dimension = self._parse_property(properties, "DIMENSION")
477            m_unit = self._parse_property(properties, "MEASURE-UNIT")
478            if dimension is not None:
479                weight_str = f"{dimension} {str(m_unit or '').lower()}".strip()
480
481        name_str = clean_desc
482        if brand:
483            name_str = re.sub(r"\b" + re.escape(brand) + r"\b", "", name_str, flags=re.IGNORECASE)
484        if weight_str:
485            name_str = re.sub(re.escape(weight_str), "", name_str, flags=re.IGNORECASE)
486            
487        name_str = re.sub(r"[,\.\:\;\_\#\-\*]", "", name_str)
488        name_str = " ".join(name_str.split()).strip().capitalize()
489        
490        if not name_str:
491            name_str = clean_desc.replace("\n", " ").strip()
492
493        price = self._parse_property(properties, "END-PRICE")
494        if price is None:
495            logger.debug(f"Skipping product '{name_str}' because final price is missing.")
496            return None
497            
498        price = float(price)
499        original_price = self._parse_property(properties, "INITIAL-PRICE")
500        if original_price is not None:
501            original_price = float(original_price)
502            if original_price <= price:
503                original_price = None
504                
505        discount_percent = self._parse_property(properties, "DISCOUNT-RATE")
506        if discount_percent is not None:
507            discount_percent = int(float(discount_percent))
508        elif original_price and original_price > 0:
509            discount_percent = int(round((1.0 - (price / original_price)) * 100))
510
511        unit_price = self._parse_property(properties, "END-KG-LT-PRICE")
512        unit_price_str = None
513        if unit_price is not None:
514            m_unit = str(self._parse_property(properties, "MEASURE-UNIT") or "KG").lower()
515            unit_price_str = f"€/{m_unit} {float(unit_price):.2f}"
516
517        promo_code = self._parse_property(properties, "DISCOUNT") or "STANDARD"
518        promo_type = "STANDARD"
519        if promo_code == "UPU" or "1+1" in clean_desc or "bogo" in clean_desc.lower():
520            promo_type = "1+1"
521        elif discount_percent:
522            promo_type = "PERCENTAGE_DISCOUNT"
523
524        start_date_raw = self._parse_property(properties, "START-DATE-VALIDITY")
525        end_date_raw = self._parse_property(properties, "END-DATE-VALIDITY")
526        validity_string = None
527        
528        if start_date_raw and end_date_raw:
529            try:
530                s_dt = datetime.strptime(str(start_date_raw)[:8], "%Y%m%d")
531                e_dt = datetime.strptime(str(end_date_raw)[:8], "%Y%m%d")
532                validity_string = f"DAL {s_dt.strftime('%d/%m/%Y')} AL {e_dt.strftime('%d/%m/%Y')}"
533            except Exception:
534                pass
535
536        category = self._parse_property(properties, "DEPARTMENT")
537
538        image_url = None
539        for prop in properties:
540            if prop.get("code") == "IMAGES":
541                vals = prop.get("values", [])
542                if vals and isinstance(vals, list):
543                    img = vals[0]
544                    u_id = img.get("uniqueId")
545                    f_name = img.get("name")
546                    if u_id and f_name:
547                        image_url = f"https://dpiu.maxidi.it/digitalflyer/api/files/{u_id}/{f_name}"
548                        break
549
550        unique_payload = f"DPIU:{store_id}:{validity_string or 'ALL'}:{name_str}:{price:.2f}"
551        offer_id = hashlib.sha256(unique_payload.encode("utf-8")).hexdigest()[:32]
552
553        return ProductOffer(
554            offer_id=offer_id,
555            supermarket="DPIU",
556            store_id=store_id,
557            name=name_str,
558            brand=brand if brand else None,
559            weight_or_volume=weight_str,
560            price=price,
561            original_price=original_price,
562            discount_percentage=discount_percent,
563            price_per_unit=unit_price_str,
564            ean_code=ean_code,
565            image_url=image_url,
566            category=category,
567            promo_type=promo_type,
568            validity_string=validity_string
569        )
